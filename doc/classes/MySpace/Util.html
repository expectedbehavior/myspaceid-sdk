    def decode_appdata_hash(hash, params = {})
      base_key = params[:base_key] || "key"
      string = String.new

      n = 0
      while true
        key = "#{base_key}#{n}"
        return unless hash.has_key?(key)
        string << hash[key]
      end

      string
    end

    def encode_appdata_hash(string, params = {}, &block)
      base_key = params[:base_key] || "key"
      batcher = Batcher.new(MAX_KEYS, {}, &block)

      chunks = string.length / MAX_VALUE_CHARS
      chunks.times do |n|
        batcher.push do |hash|
          hash.update("#{base_key}#{n}" => string[n * MAX_VALUE_CHARS, MAX_VALUE_CHARS])
        end
      end
      
      batcher.flush
    end
